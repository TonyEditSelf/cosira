        // OKLCH to RGB conversion (simplified for visualization)
        function oklchToRgb(l, c, h) {
            const a = c * Math.cos(h * Math.PI / 180);
            const b = c * Math.sin(h * Math.PI / 180);
            
            // Simplified conversion (not accurate, just for visualization)
            let r = l + 0.4 * a + 0.2 * b;
            let g = l - 0.2 * a + 0.3 * b;
            let blue = l - 0.4 * a - 0.5 * b;
            
            r = Math.max(0, Math.min(1, r)) * 255;
            g = Math.max(0, Math.min(1, g)) * 255;
            blue = Math.max(0, Math.min(1, blue)) * 255;
            
            return `rgb(${Math.round(r)}, ${Math.round(g)}, ${Math.round(blue)})`;
        }

        // Draw color wheel arc
        function drawColorWheelArc(canvasId, hueSpan, numColors = 10) {
            const canvas = document.getElementById(canvasId);
            const ctx = canvas.getContext('2d');
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;
            const radius = 150;
            const baseHue = 30;
            const baseL = 0.65;
            const baseC = 0.15;
            
            // Draw wheel background
            ctx.fillStyle = '#f8f9fa';
            ctx.fillRect(0, 0, canvas.width, canvas.height);
            
            // Draw color wheel
            for (let angle = 0; angle < 360; angle += 2) {
                ctx.beginPath();
                ctx.arc(centerX, centerY, radius, angle * Math.PI / 180, (angle + 2) * Math.PI / 180);
                ctx.strokeStyle = oklchToRgb(baseL, baseC, angle);
                ctx.lineWidth = 20;
                ctx.stroke();
            }
            
            // Draw arc span
            const startAngle = baseHue - hueSpan / 2;
            const endAngle = baseHue + hueSpan / 2;
            
            ctx.beginPath();
            ctx.arc(centerX, centerY, radius, startAngle * Math.PI / 180, endAngle * Math.PI / 180);
            ctx.strokeStyle = '#667eea';
            ctx.lineWidth = 25;
            ctx.stroke();
            
            // Draw color points
            for (let i = 0; i < numColors; i++) {
                const t = i / (numColors - 1);
                const hueOffset = (t - 0.5) * hueSpan;
                const hue = baseHue + hueOffset;
                const angle = hue * Math.PI / 180;
                
                const x = centerX + radius * Math.cos(angle);
                const y = centerY + radius * Math.sin(angle);
                
                ctx.beginPath();
                ctx.arc(x, y, 12, 0, 2 * Math.PI);
                ctx.fillStyle = oklchToRgb(baseL, baseC, hue);
                ctx.fill();
                ctx.strokeStyle = 'white';
                ctx.lineWidth = 3;
                ctx.stroke();
                
                ctx.strokeStyle = '#333';
                ctx.lineWidth = 1;
                ctx.stroke();
            }
            
            // Label
            ctx.fillStyle = '#333';
            ctx.font = 'bold 16px sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText(`Hue span: ${hueSpan}°`, centerX, 30);
            ctx.font = '14px sans-serif';
            ctx.fillText(`${numColors} colors`, centerX, 50);
        }

        // Draw LC plane arc
        function drawLCPlaneArc(canvasId, arcType) {
            const canvas = document.getElementById(canvasId);
            const ctx = canvas.getContext('2d');
            const width = canvas.width;
            const height = canvas.height;
            const baseL = 0.6;
            const baseC = 0.15;
            const baseH = 220; // Blue
            const numColors = 10;
            
            // Clear
            ctx.fillStyle = '#f8f9fa';
            ctx.fillRect(0, 0, width, height);
            
            // Draw axes
            const margin = 50;
            const plotWidth = width - 2 * margin;
            const plotHeight = height - 2 * margin;
            
            // Axes
            ctx.strokeStyle = '#ccc';
            ctx.lineWidth = 2;
            ctx.beginPath();
            ctx.moveTo(margin, height - margin);
            ctx.lineTo(width - margin, height - margin);
            ctx.moveTo(margin, height - margin);
            ctx.lineTo(margin, margin);
            ctx.stroke();
            
            // Labels
            ctx.fillStyle = '#666';
            ctx.font = '14px sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText('Chroma (C)', width / 2, height - 10);
            ctx.save();
            ctx.translate(15, height / 2);
            ctx.rotate(-Math.PI / 2);
            ctx.fillText('Lightness (L)', 0, 0);
            ctx.restore();
            
            // Convert L,C to canvas coordinates
            const toX = (c) => margin + (c / 0.4) * plotWidth;
            const toY = (l) => height - margin - (l * plotHeight);
            
            // Draw path based on type
            let points = [];
            
            if (arcType === 'circular') {
                const radius = 0.12;
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const angle = (-90 + t * 180) * Math.PI / 180;
                    const l = baseL + radius * Math.sin(angle);
                    const c = baseC + radius * Math.cos(angle);
                    points.push({ l, c });
                }
            } else if (arcType === 'lightness') {
                const range = 0.5;
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const l = baseL + (t - 0.5) * range;
                    points.push({ l, c: baseC });
                }
            } else if (arcType === 'chroma') {
                const range = 0.2;
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const c = baseC + (t - 0.5) * range;
                    points.push({ l: baseL, c });
                }
            } else if (arcType === 'diagonal') {
                const lRange = 0.4;
                const cRange = 0.15;
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const offset = t - 0.5;
                    const l = baseL + offset * lRange;
                    const c = baseC + offset * cRange;
                    points.push({ l, c });
                }
            }
            
            // Draw path
            ctx.strokeStyle = '#667eea';
            ctx.lineWidth = 3;
            ctx.beginPath();
            points.forEach((p, i) => {
                const x = toX(p.c);
                const y = toY(p.l);
                if (i === 0) ctx.moveTo(x, y);
                else ctx.lineTo(x, y);
            });
            ctx.stroke();
            
            // Draw points
            points.forEach((p, i) => {
                const x = toX(p.c);
                const y = toY(p.l);
                
                ctx.beginPath();
                ctx.arc(x, y, 10, 0, 2 * Math.PI);
                ctx.fillStyle = oklchToRgb(p.l, p.c, baseH);
                ctx.fill();
                ctx.strokeStyle = 'white';
                ctx.lineWidth = 3;
                ctx.stroke();
                ctx.strokeStyle = '#333';
                ctx.lineWidth = 1;
                ctx.stroke();
            });
            
            // Mark base point
            ctx.beginPath();
            ctx.arc(toX(baseC), toY(baseL), 5, 0, 2 * Math.PI);
            ctx.fillStyle = 'red';
            ctx.fill();
        }

        // Draw 3D arc visualization
        function draw3DArc(canvasId, arcType) {
            const canvas = document.getElementById(canvasId);
            const ctx = canvas.getContext('2d');
            const width = canvas.width;
            const height = canvas.height;
            const numColors = 10;
            
            ctx.fillStyle = '#f8f9fa';
            ctx.fillRect(0, 0, width, height);
            
            // Draw 3D visualization (simplified projection)
            const centerX = width / 2;
            const centerY = height / 2;
            
            let points = [];
            
            if (arcType === 'spiral') {
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const offset = t - 0.5;
                    const l = 0.6 + offset * 0.3;
                    const c = 0.15 + offset * 0.1;
                    const h = 30 + t * 120;
                    points.push({ l, c, h });
                }
            } else if (arcType === 'helix') {
                for (let i = 0; i < numColors; i++) {
                    const t = i / (numColors - 1);
                    const angle = t * Math.PI * 2;
                    const l = 0.6 + 0.1 * Math.sin(angle);
                    const c = 0.15 + 0.1 * Math.cos(angle);
                    const h = 30 + t * 180;
                    points.push({ l, c, h });
                }
            }
            
            // Draw 3D spiral path
            ctx.strokeStyle = '#667eea';
            ctx.lineWidth = 3;
            ctx.beginPath();
            
            points.forEach((p, i) => {
                // Project 3D to 2D
                const radius = 100 + p.l * 100;
                const angle = p.h * Math.PI / 180;
                const x = centerX + radius * Math.cos(angle);
                const y = centerY + radius * Math.sin(angle) - (p.c * 200);
                
                if (i === 0) ctx.moveTo(x, y);
                else ctx.lineTo(x, y);
            });
            ctx.stroke();
            
            // Draw color points
            points.forEach((p, i) => {
                const radius = 100 + p.l * 100;
                const angle = p.h * Math.PI / 180;
                const x = centerX + radius * Math.cos(angle);
                const y = centerY + radius * Math.sin(angle) - (p.c * 200);
                
                ctx.beginPath();
                ctx.arc(x, y, 12, 0, 2 * Math.PI);
                ctx.fillStyle = oklchToRgb(p.l, p.c, p.h);
                ctx.fill();
                ctx.strokeStyle = 'white';
                ctx.lineWidth = 3;
                ctx.stroke();
                ctx.strokeStyle = '#333';
                ctx.lineWidth = 1;
                ctx.stroke();
            });
            
            // Label
            ctx.fillStyle = '#333';
            ctx.font = 'bold 14px sans-serif';
            ctx.textAlign = 'center';
            ctx.fillText('L, C, and H all changing', centerX, 30);
        }

        // Draw all visualizations
        window.addEventListener('load', () => {
            drawColorWheelArc('hue-canvas', 180);
            drawColorWheelArc('analogous-canvas', 45);
            drawColorWheelArc('complementary-canvas', 180);
            
            drawLCPlaneArc('circular-canvas', 'circular');
            drawLCPlaneArc('lightness-canvas', 'lightness');
            drawLCPlaneArc('chroma-canvas', 'chroma');
            drawLCPlaneArc('diagonal-canvas', 'diagonal');
            
            draw3DArc('spiral-canvas', 'spiral');
            draw3DArc('helix-canvas', 'helix');
        });
