const WCAG_AA_RATIO = 4.5;

const getLuminance = (hex) => {
  if (!hex || typeof hex !== "string" || !hex.startsWith("#")) {
    return 0;
  }
  const rgb = parseInt(hex.slice(1), 16);
  const r = (rgb >> 16) & 0xff;
  const g = (rgb >> 8) & 0xff;
  const b = (rgb >> 0) & 0xff;
  const [rs, gs, bs] = [r, g, b].map((c) => {
    c = c / 255;
    return c <= 0.03928 ? c / 12.92 : Math.pow((c + 0.055) / 1.055, 2.4);
  });
  return 0.2126 * rs + 0.7152 * gs + 0.0722 * bs;
};

const getContrastRatio = (color1, color2) => {
  const l1 = getLuminance(color1);
  const l2 = getLuminance(color2);
  const lighter = Math.max(l1, l2);
  const darker = Math.min(l1, l2);
  return (lighter + 0.05) / (darker + 0.05);
};

const meetsWCAGAA = (bgColor, fgColor) => {
  return getContrastRatio(bgColor, fgColor) >= WCAG_AA_RATIO;
};

const getContrastingColors = (color, palette) => {
  return palette.filter((c) => meetsWCAGAA(color, c));
};

// Editable Element Component
const EditableElement = ({
  children,
  type,
  id,
  onSelect,
  isSelected,
  styles,
  className = "",
}) => {
  const [isHovered, setIsHovered] = useState(false);

  const handleClick = (e) => {
    e.stopPropagation();
    onSelect(id, type, styles);
  };

  const borderClass = isSelected
    ? "outline outline-4 outline-blue-500"
    : isHovered
    ? "outline outline-2 outline-blue-300"
    : "";

  return (
    <div
      className={`relative ${borderClass} ${className}`}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
      onClick={handleClick}
      style={styles}
    >
      {children}
      {isHovered && !isSelected && (
        <div className="absolute -top-8 left-0 bg-gray-900 text-white text-xs px-2 py-1 rounded shadow-lg z-50 whitespace-nowrap">
          {type === "text" && styles?.fontFamily && (
            <>
              Font: {styles.fontFamily} | Size: {styles.fontSize || "16px"}
            </>
          )}
          {type === "color" && styles?.backgroundColor && (
            <>Color: {styles.backgroundColor}</>
          )}
          {type === "body" && <>Click to edit background</>}
        </div>
      )}
    </div>
  );
};

// Element-specific Toolbar Component (appears above permanent toolbar)
const ElementToolbar = ({
  selectedElement,
  onClose,
  onStyleChange,
  palette,
}) => {
  const [showFonts, setShowFonts] = useState(false);
  const [showColors, setShowColors] = useState(false);
  const [colorType, setColorType] = useState(null); // 'text' or 'background'

  if (!selectedElement) return null;

  const handleFontChange = (font) => {
    onStyleChange({ ...selectedElement.styles, fontFamily: font });
    setShowFonts(false);
  };

  const handleColorChange = (color) => {
    if (colorType === "text") {
      onStyleChange({ ...selectedElement.styles, color });
    } else {
      onStyleChange({ ...selectedElement.styles, backgroundColor: color });
    }
  };

  return (
    <div
      className="fixed left-0 right-0 bg-gray-800 border-t-4 border-blue-500 shadow-2xl z-40"
      style={{ bottom: "200px" }}
    >
      <div className="flex items-center justify-between px-6 py-4">
        <div className="flex items-center gap-4">
          <span className="text-white font-semibold">
            Editing: {selectedElement.type}
          </span>

          {selectedElement.type === "text" && (
            <>
              <button
                onClick={() => setShowFonts(!showFonts)}
                className="bg-blue-600 hover:bg-blue-700 text-white px-4 py-2 rounded flex items-center gap-2"
              >
                <Palette size={16} />
                Typography
              </button>

              <button
                onClick={() => {
                  setColorType("text");
                  setShowColors(!showColors);
                }}
                className="bg-purple-600 hover:bg-purple-700 text-white px-4 py-2 rounded"
              >
                Text Color
              </button>

              <button
                onClick={() => {
                  setColorType("background");
                  setShowColors(!showColors);
                }}
                className="bg-green-600 hover:bg-green-700 text-white px-4 py-2 rounded"
              >
                Background
              </button>
            </>
          )}

          {selectedElement.type === "body" && (
            <button
              onClick={() => {
                setColorType("background");
                setShowColors(!showColors);
              }}
              className="bg-green-600 hover:bg-green-700 text-white px-4 py-2 rounded"
            >
              Background Color
            </button>
          )}
        </div>

        <button onClick={onClose} className="text-white hover:text-gray-300">
          <X size={24} />
        </button>
      </div>

      {/* Font Selector */}
      {showFonts && (
        <div className="bg-gray-700 border-t border-gray-600 p-4 max-h-64 overflow-y-auto">
          <div className="grid grid-cols-4 gap-2">
            {GOOGLE_FONTS.map((font) => (
              <button
                key={font}
                onClick={() => handleFontChange(font)}
                className="bg-gray-600 hover:bg-gray-500 text-white px-3 py-2 rounded text-sm text-left"
                style={{ fontFamily: font }}
              >
                {font}
              </button>
            ))}
          </div>
        </div>
      )}

      {/* Color Picker */}
      {showColors && (
        <div className="bg-gray-700 border-t border-gray-600 p-4">
          <div className="grid grid-cols-12 gap-2 max-w-3xl">
            {palette.map((color, i) => (
              <button
                key={i}
                onClick={() => handleColorChange(color)}
                className="aspect-square rounded border-2 border-gray-600 hover:border-white transition-colors"
                style={{ backgroundColor: color }}
                title={color}
              />
            ))}
          </div>
        </div>
      )}
    </div>
  );
};

          <span className="text-gray-500 text-xs">
            Click any element above to edit • {palette.length} colors available
          </span>
        </div>
      </div>
    </div>
  );
};

  const hexPalette = useMemo(() => {
    return (
      palette?.map((colorObj) => {
        const { l, c, h, a } = colorObj.value;
        return oklchToHex(l, c, h, a);
      }) || []
    );
  }, [palette]);

  const fullPalette = useMemo(() => {
    return [...hexPalette, "#000000", "#FFFFFF", "#808080"];
  }, [hexPalette]);

  const [fontsLoaded, setFontsLoaded] = useState(false);

  const [styles, setStyles] = useState({

      color: hexPalette[0] || "#000",
    },
    footer: { fontFamily: "Open Sans", fontSize: "14px", color: "#666" },

  useEffect(() => {
    const allFonts = Object.values(styles)
      .map((s) => s.fontFamily)

                styles={styles.logo}
                className="flex items-center gap-3 cursor-pointer"

                />

                  type="text"
                  id="hero"
                  onSelect={handleSelect}
                  isSelected={selectedElement?.id === "hero"}
                  styles={styles.hero}
                  className="cursor-pointer mb-6"

                      Design
